  <script>
    // ========== BASEROW Konfiguration (AKTUALISIERT) ==========
    const BASEROW_API_TOKEN = "SvWXJIjz6eyKd1Z7REyRpCtnuCqJYPrQ"; 
    const BASEROW_SERVER_URL = "https://api.baserow.io";
    const EINTRAEGE_TABLE_ID = 991494; // Deine neue ID für Einträge
    const KONFIG_TABLE_ID    = 991493; // Deine neue ID für Konfig
    // ==========================================================

    let admin = false;
    let aktuellerTag = 'Aufbau';
    let isLoadingData = false;

    const daten = { "Aufbau": {}, "Freitag": {}, "Samstag": {}, "Sonntag": {}, "Abbau": {} };
    const config = { 
      "Aufbau": { aufgaben: [], zeiten: [], farben: [], limits: {} }, 
      "Freitag": { aufgaben: [], zeiten: [], farben: [], limits: {} }, 
      "Samstag": { aufgaben: [], zeiten: [], farben: [], limits: {} }, 
      "Sonntag": { aufgaben: [], zeiten: [], farben: [], limits: {} }, 
      "Abbau": { aufgaben: [], zeiten: [], farben: [], limits: {} }
    };

    async function fetchBaserow(method, url, body = null) {
      const headers = { 'Authorization': `Token ${BASEROW_API_TOKEN}`, 'Content-Type': 'application/json' };
      const options = { method, headers };
      if (body) options.body = JSON.stringify(body);
      const response = await fetch(url, options);
      if (!response.ok) throw new Error(`API Fehler: ${response.status}`);
      return response.status === 204 ? {success:true} : await response.json();
    }

    async function ladeDaten() {
      if (isLoadingData) return;
      isLoadingData = true;
      try {
        let allRows = [];
        let nextUrl = `${BASEROW_SERVER_URL}/api/database/rows/table/${EINTRAEGE_TABLE_ID}/?user_field_names=true&size=100`;
        
        while(nextUrl) {
          const pageData = await fetchBaserow('GET', nextUrl);
          allRows = allRows.concat(pageData.results);
          nextUrl = pageData.next ? pageData.next.replace('http://', 'https://') : null;
        }

        for (const t in daten) daten[t] = {};
        for (const row of allRows) {
          if (!row.Tag || !row.Zeit || !row.Aufgabe || !row.Name) continue;
          const t = String(row.Tag).trim();
          const key = `${String(row.Zeit).trim()}||${String(row.Aufgabe).trim()}`;
          if (daten[t]) {
            if (!daten[t][key]) daten[t][key] = [];
            daten[t][key].push(String(row.Name).trim());
          }
        }
        await ladeKonfiguration();
        zeigeTag(aktuellerTag);
      } catch (e) {
        console.error("Ladefehler:", e);
      } finally {
        isLoadingData = false;
      }
    }

    async function ladeKonfiguration() {
      const url = `${BASEROW_SERVER_URL}/api/database/rows/table/${KONFIG_TABLE_ID}/?user_field_names=true`;
      try {
        const data = await fetchBaserow('GET', url);
        data.results.forEach(row => {
          if (row.Tag && row.Konfig && config[row.Tag]) {
            config[row.Tag] = JSON.parse(row.Konfig);
          }
        });
      } catch (e) { console.error("Konfig-Fehler:", e); }
    }

    function zeigeTag(t) {
      aktuellerTag = t;
      const conf = config[t];
      const container = document.getElementById('schichtplan');
      
      if (!conf || !conf.aufgaben?.length) {
        container.innerHTML = `<h2>${t} 11.06. - 15.06.</h2><p style="text-align:center;">Noch keine Aufgaben für diesen Tag konfiguriert.<br>Nutze den Admin-Login zum Einrichten.</p>`;
        return;
      }

      let html = `<h2>${t}</h2><div class="scroll-container"><table><thead><tr><th>Aufgabe</th>`;
      conf.zeiten.forEach(z => html += `<th>${z}</th>`);
      html += `</tr></thead><tbody>`;

      conf.aufgaben.forEach((a, idx) => {
        html += `<tr><th style="background:${conf.farben[idx]}">${a}</th>`;
        conf.zeiten.forEach(z => {
          const key = `${z}||${a}`;
          const helfer = daten[t][key] || [];
          const max = conf.limits[key] || 3;
          const istVoll = helfer.length >= max;

          html += `<td><strong>${helfer.length}/${max}</strong><ul class="helfer-list">`;
          helfer.forEach(n => html += `<li>${n}</li>`);
          html += `</ul>`;
          
          if (istVoll) {
            html += `<div class="schicht-voll">✔️ Voll</div>`;
          } else {
            html += `<div class="input-gruppe">
              <input id="name-${key}" placeholder="Name">
              <button onclick="eintragen('${key}')">✔️</button>
            </div>`;
          }
          html += `</td>`;
        });
        html += `</tr>`;
      });

      html += `</tbody></table></div>`;
      container.innerHTML = html;
      if(admin) zeigeDebug();
    }

    async function eintragen(key) {
      const input = document.getElementById('name-' + key);
      const name = input.value.trim();
      if (!name) return;
      const [zeit, aufgabe] = key.split('||');
      
      try {
        await fetchBaserow('POST', `${BASEROW_SERVER_URL}/api/database/rows/table/${EINTRAEGE_TABLE_ID}/?user_field_names=true`, {
          "Tag": aktuellerTag, "Zeit": zeit, "Aufgabe": aufgabe, "Name": name
        });
        input.value = "";
        await ladeDaten();
      } catch (e) { alert("Fehler beim Speichern!"); }
    }

    function zeigeLogin() { document.getElementById('loginFeld').style.display = 'block'; }
    function adminLogin() {
      if (document.getElementById('adminPasswort').value === 'admin') {
        admin = true;
        document.getElementById('adminBereich').style.display = 'block';
        document.getElementById('loginFeld').style.display = 'none';
        zeigeTag(aktuellerTag);
      } else { alert("Falsch!"); }
    }

    async function speichereKonfiguration() {
      const getUrl = `${BASEROW_SERVER_URL}/api/database/rows/table/${KONFIG_TABLE_ID}/?user_field_names=true`;
      const data = await fetchBaserow('GET', getUrl);
      const existing = data.results.find(r => r.Tag === aktuellerTag);
      const payload = { "Konfig": JSON.stringify(config[aktuellerTag]) };

      if (existing) {
        await fetchBaserow('PATCH', `${BASEROW_SERVER_URL}/api/database/rows/table/${KONFIG_TABLE_ID}/${existing.id}/?user_field_names=true`, payload);
      } else {
        await fetchBaserow('POST', `${BASEROW_SERVER_URL}/api/database/rows/table/${KONFIG_TABLE_ID}/?user_field_names=true`, { "Tag": aktuellerTag, ...payload });
      }
      await ladeDaten();
    }

    async function aufgabeHinzufuegen() {
      const a = document.getElementById('aufgabe').value.trim();
      const f = document.getElementById('farbe').value;
      if (a) { 
        config[aktuellerTag].aufgaben.push(a); 
        config[aktuellerTag].farben.push(f); 
        await speichereKonfiguration(); 
        document.getElementById('aufgabe').value = "";
      }
    }

    async function zeitHinzufuegen() {
      const z = document.getElementById('zeit').value.trim();
      if (z) { 
        config[aktuellerTag].zeiten.push(z); 
        await speichereKonfiguration(); 
        document.getElementById('zeit').value = "";
      }
    }

    async function zeitLoeschen() {
      config[aktuellerTag].zeiten.pop();
      await speichereKonfiguration();
    }

    async function loeschen() {
      config[aktuellerTag].aufgaben.pop();
      config[aktuellerTag].farben.pop();
      await speichereKonfiguration();
    }

    async function maxHinzufuegen() {
      const z = document.getElementById('maxZeit').value.trim();
      const a = document.getElementById('maxAufgabe').value.trim();
      const m = parseInt(document.getElementById('max').value);
      if (z && a && !isNaN(m)) {
        config[aktuellerTag].limits[`${z}||${a}`] = m;
        await speichereKonfiguration();
      }
    }

    function zeigeVerantwortliche() { document.getElementById('verantwortlicheBox').style.display = 'block'; }
    function verantwortlicheSchliessen() { document.getElementById('verantwortlicheBox').style.display = 'none'; }

    function zeigeDebug() {
      const db = document.getElementById('debug-block');
      db.style.display = 'block';
      db.textContent = JSON.stringify(config[aktuellerTag], null, 2);
    }

    ladeDaten();
  </script>
